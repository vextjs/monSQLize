**API**:
```javascript
// 定义 Saga
msq.defineSaga({
    name: 'create-order-with-payment',
    steps: [
        {
            name: 'create-order',
            execute: async (ctx) => {
                const order = await createOrder(ctx.data);
                ctx.set('orderId', order.id);
                return order;
            },
            compensate: async (ctx) => {
                await cancelOrder(ctx.get('orderId'));
            }
        },
        {
            name: 'charge-payment',
            execute: async (ctx) => {
                const charge = await stripe.charges.create({
                    amount: ctx.data.amount,
                    source: ctx.data.paymentToken
                });
                ctx.set('chargeId', charge.id);
                return charge;
            },
            compensate: async (ctx) => {
                await stripe.refunds.create({ 
                    charge: ctx.get('chargeId') 
                });
            }
        }
    ]
});

// 执行 Saga
const result = await msq.executeSaga('create-order-with-payment', {
    userId: 'user123',
    amount: 9900,
    paymentToken: 'tok_visa'
});
```

### 2. 企业级多连接池管理

**核心能力**:
- ✅ 支持多个 MongoDB 连接池（primary、secondary、analytics、custom）
- ✅ 5种智能选择策略（auto、roundRobin、weighted、leastConnections、manual）
- ✅ 实时健康检查和自动故障转移
- ✅ 完整的统计收集和监控
- ✅ 灵活的配置和扩展性

**新增模块**:

#### ConnectionPoolManager
统一管理多个连接池，提供智能路由和故障转移。

```javascript
const manager = new ConnectionPoolManager({
    maxPoolsCount: 10,
    poolStrategy: 'auto',
    fallback: {
        enabled: true,
        fallbackStrategy: 'readonly'
    }
});

// 添加主库
await manager.addPool({
    name: 'primary',
    uri: 'mongodb://primary:27017/db',
    role: 'primary',
    weight: 1
});

// 添加只读副本
await manager.addPool({
    name: 'secondary',
    uri: 'mongodb://secondary:27017/db',
    role: 'secondary',
    weight: 2
});

// 智能选择
const pool = manager.selectPool('read');  // 自动选择最佳池
```

#### HealthChecker
实时监控连接池健康状态，支持自动恢复。

```javascript
const checker = new HealthChecker({
    logger: console
});

checker.register({
    name: 'primary',
    healthCheck: {
        enabled: true,
        interval: 5000,    // 5秒检查一次
        timeout: 3000,     // 3秒超时
        retries: 3         // 失败重试3次
    }
}, client);

checker.start();

// 查询健康状态
const status = checker.getStatus('primary');
console.log(status);  // { status: 'up', consecutiveFailures: 0, ... }
```

#### PoolSelector
5种智能选择策略，满足不同场景需求。

```javascript
const selector = new PoolSelector({
    strategy: 'auto'  // auto | roundRobin | weighted | leastConnections | manual
});

// 自动策略：write → primary, read → secondary
const selected = selector.select(pools, { operation: 'read' });
```

#### PoolStats
完整的统计收集，支持性能分析。

```javascript
const stats = new PoolStats({
    batchSize: 100,
    batchInterval: 1000
});

// 记录查询
await stats.recordQuery('primary', 150, null);

// 获取统计
const poolStats = stats.getStats('primary');
console.log(poolStats);
// {
//   totalRequests: 1000,
//   avgResponseTime: 45,
//   minResponseTime: 10,
//   maxResponseTime: 200,
//   errorRate: 0.01
// }
```

**配置示例**:

```javascript
const { ConnectionPoolManager } = require('monsqlize');

const manager = new ConnectionPoolManager({
    maxPoolsCount: 10,
    poolStrategy: 'auto',
    fallback: {
        enabled: true,
        fallbackStrategy: 'readonly',
        retryDelay: 1000,
        maxRetries: 3
    },
    logger: console
});

// 配置主库
await manager.addPool({
    name: 'primary',
    uri: process.env.MONGODB_PRIMARY_URI,
    role: 'primary',
    weight: 1,
    tags: ['production', 'main'],
    options: {
        maxPoolSize: 100,
        minPoolSize: 10,
        maxIdleTimeMS: 30000
    },
    healthCheck: {
        enabled: true,
        interval: 5000,
        timeout: 3000,
        retries: 3
    }
});

// 配置只读副本
await manager.addPool({
    name: 'secondary-1',
    uri: process.env.MONGODB_SECONDARY_URI,
    role: 'secondary',
    weight: 2,
    tags: ['production', 'read'],
    healthCheck: {
        enabled: true,
        interval: 10000
    }
});

// 配置分析库
await manager.addPool({
    name: 'analytics',
    uri: process.env.MONGODB_ANALYTICS_URI,
    role: 'analytics',
    weight: 1,
    tags: ['analytics', 'reports']
});

// 启动健康检查
manager.startHealthCheck();

// 使用示例
const pool = manager.selectPool('read', { tags: ['production'] });
const results = await pool.collection.find({ status: 'active' }).toArray();

// 获取所有池的统计
const allStats = manager.getPoolStats();
console.log(allStats);
```

### 2. Update 方法支持聚合管道

**核心能力**:
- ✅ `updateOne` 和 `updateMany` 支持聚合管道语法
- ✅ 字段间计算（`$add`、`$multiply`、`$subtract`）
- ✅ 条件赋值（`$cond`、`$ifNull`）
- ✅ 多阶段转换（`$concat`、`$toLower`、`$toUpper`）
- ✅ 完全兼容 MongoDB 4.2+ 语法

**使用示例**:

```javascript
// 字段间计算
await collection.updateOne(
    { _id: userId },
    [
        {
            $set: {
                totalPrice: { $add: ['$price', '$tax'] },
                discount: { $multiply: ['$price', 0.1] }
            }
        }
    ]
);

// 条件赋值
await collection.updateMany(
    { status: 'pending' },
    [
        {
            $set: {
                priority: {
                    $cond: {
                        if: { $gte: ['$amount', 1000] },
                        then: 'high',
                        else: 'normal'
                    }
                }
            }
        }
    ]
);

// 多阶段转换
await collection.updateOne(
    { _id: userId },
    [
        {
            $set: {
                fullName: { $concat: ['$firstName', ' ', '$lastName'] },
                email: { $toLower: '$email' }
            }
        },
        {
            $set: {
                displayName: { $ifNull: ['$nickname', '$fullName'] }
            }
        }
    ]
);
```

**技术实现**:
- 自动检测聚合管道语法（数组格式）
- 完整的参数验证
- 与普通更新语法完全兼容
- 详细的错误提示

**API 设计（已完成）**:

```javascript
const { SagaOrchestrator } = require('monsqlize');

const saga = new SagaOrchestrator({
    timeout: 30000,
    logger: console
});

// 定义 Saga 步骤
const orderSaga = saga.define('createOrder')
    .step('reserveInventory', {
        action: async (ctx) => {
            const result = await inventoryService.reserve(ctx.productId, ctx.quantity);
            ctx.reservationId = result.id;
            return result;
        },
        compensate: async (ctx) => {
            await inventoryService.release(ctx.reservationId);
        }
    })
    .step('createPayment', {
        action: async (ctx) => {
            const payment = await paymentService.charge(ctx.userId, ctx.amount);
            ctx.paymentId = payment.id;
            return payment;
        },
        compensate: async (ctx) => {
            await paymentService.refund(ctx.paymentId);
        }
    })
    .step('createOrder', {
        action: async (ctx) => {
            return await orderService.create({
                userId: ctx.userId,
                productId: ctx.productId,
                quantity: ctx.quantity,
                paymentId: ctx.paymentId
            });
        },
        compensate: async (ctx) => {
            await orderService.cancel(ctx.orderId);
        }
    });

// 执行 Saga
try {
    const result = await orderSaga.execute({
        userId: 'user123',
        productId: 'prod456',
        quantity: 2,
        amount: 199.99
    });
    console.log('Order created:', result);
} catch (error) {
    console.error('Saga failed, compensated:', error);
}
```

**特性设计**:
- 自动补偿：任何步骤失败，自动回滚已执行的步骤
- 状态跟踪：完整记录每个步骤的执行状态
- 超时处理：支持全局和单步超时
- 重试机制：支持失败重试
- 日志记录：完整的执行日志
